
* Generic stuff

  #+name: intellij-style auto-save
  #+begin_src emacs-lisp
  (setq auto-save-interval 1
        auto-save-timeout  1)

  (defun full-auto-save ()
    (interactive)
    (save-excursion
      (dolist (buf (buffer-list))
        (set-buffer buf)
        (if (and (buffer-file-name) (buffer-modified-p))
            (basic-save-buffer)))))

  (add-hook 'auto-save-hook 'full-auto-save)
  #+end_src
